銷售總額與淨利:

    月線上銷售總額 = {
        抓 13.銷售紀錄表.銷售方式 == 1, 13.銷售日期 >= 本月第一天, 銷售日期 <= 本月最後一天
            sum += 13.產品編號(FK to 5.產品表格).產品價格
        }

    月線上淨利 = {
        抓 13.銷售紀錄表.銷售方式 == 1, 13.銷售日期 >= 本月第一天, 銷售日期 <= 本月最後一天
            sum += 13.產品編號(FK to 5.產品表格).產品價格 - 13.產品編號(FK to 5.產品表格).產品成本
        }

    月實體店號總銷售總額 = {
        抓 13.銷售紀錄表.銷售方式 == 2, 13.銷售日期 >= 本月第一天, 銷售日期 <= 本月最後一天, 13.店號 == 要求的店號
            sum += 13.業務員售出價格
        }

    月實體店面銷售總額 = {
        抓 13.銷售紀錄表.銷售方式 == 2, 13.銷售日期 >= 本月第一天, 銷售日期 <= 本月最後一天
            sum += 13.業務員售出價格
        },
        {
            sum += 所有 月實體店號總銷售總額
        }

    月實體店號淨利 = {
        抓 13.銷售紀錄表.銷售方式 == 2, 13.銷售日期 >= 本月第一天, 銷售日期 <= 本月最後一天, 13.店號 == 要求的店號
            sum += 13.業務員售出價格 - 13.產品編號(FK to 5.產品表格).產品成本
        }

    月實體店面總淨利 = {
        抓 13.銷售紀錄表.銷售方式 == 2, 13.銷售日期 >= 本月第一天, 銷售日期 <= 本月最後一天
            sum += 13.業務員售出價格 - 13.產品編號(FK to 5.產品表格).產品成本
        },
        {
            sum += 所有 月實體店號淨利
        }

    月總銷售總額 = 月線上銷售總額 + 月實體店面銷售總額

    月總淨利 = 月線上淨利 + 月實體店面淨利
